// Misión 1 · Componente: Pulsador

Confirmar el Pronóstico — función con un solo parámetro

El pulsador que armaron en Tinkercad ahora es el botón "Enviar" del Prode: lo presionás para bloquear tu pronóstico antes de que arranque el partido.

Contexto TinkercadRecordá tu circuito

El pulsador devuelve HIGH (1) cuando está presionado y LOW (0) cuando no. En tu simulación lo leían con digitalRead(pinPulsador). Hoy lo vamos a envolver en una función C++.

🔶
Función con 1 parámetro
String confirmarPronostico(int estadoPulsador)
const int PIN_PULSADOR = 2; String confirmarPronostico(int estadoPulsador) { // Recibe el estado del pulsador (0 o 1) y devuelve si el pronóstico ya quedó cerrado if (estadoPulsador == 1) { return "PRONÓSTICO CONFIRMADO"; } else { return "Esperando que confirmes..."; } } void setup() { Serial.begin(9600); pinMode(PIN_PULSADOR, INPUT); } void loop() { int lectura = digitalRead(PIN_PULSADOR); String estado = confirmarPronostico(lectura); Serial.println(estado); delay(500); }
import pyfirmata import time board = pyfirmata.Arduino('COM3') pin_pulsador = board.get_pin('d:2:i') def confirmar_pronostico(estado_pulsador): # Recibe el estado del pulsador (0 o 1) y devuelve si el pronóstico ya quedó cerrado if estado_pulsador == 1: return "PRONÓSTICO CONFIRMADO" else: return "Esperando que confirmes..." it = pyfirmata.util.Iterator(board) it.start() pin_pulsador.enable_reporting() while True: lectura = pin_pulsador.read() estado = confirmar_pronostico(1 if lectura else 0) print(estado) time.sleep(0.5)
Conectá los puntos: confirmarPronostico / confirmar_pronostico es la misma lógica if/else de siempre, solo que ahora vive adentro de una función con nombre propio, lista para llamarse cada vez que haga falta.

Pronosticar el Resultado — función con valores que cambian

El potenciómetro que giraban en Tinkercad ahora fija cuántos goles pronosticás para cada equipo, antes de que arranque el partido.

Contexto TinkercadRecordá tu circuito

El potenciómetro devuelve un valor analógico entre 0 y 1023 con analogRead(). Para este ejercicio simplificamos: vamos a mapear ese valor a un pronóstico de goles entre 0 y 5 con map().

🎯
Función con return numérico
int leerPronostico(int valorPotenciometro)
const int PIN_POT_LOCAL = A0; const int PIN_POT_VISITANTE = A1; int leerPronostico(int valorPotenciometro) { // Convierte una lectura 0–1023 del potenciómetro en un pronóstico de goles (0 a 5) int goles = map(valorPotenciometro, 0, 1023, 0, 5); return goles; } String mostrarPronostico(int golesLocal, int golesVisitante) { // Recibe el pronóstico de ambos equipos y arma el mensaje a mostrar String ganador; if (golesLocal > golesVisitante) { ganador = "ganador: Local"; } else if (golesVisitante > golesLocal) { ganador = "ganador: Visitante"; } else { ganador = "empate"; } return "Tu pronóstico: Local " + String(golesLocal) + " - " + String(golesVisitante) + " Visitante | " + ganador; } void loop() { int potLocal = analogRead(PIN_POT_LOCAL); int potVisitante = analogRead(PIN_POT_VISITANTE); int golesL = leerPronostico(potLocal); int golesV = leerPronostico(potVisitante); Serial.println(mostrarPronostico(golesL, golesV)); delay(1000); }
import pyfirmata import time board = pyfirmata.Arduino('COM3') pot_local = board.get_pin('a:0:i') pot_visitante = board.get_pin('a:1:i') def leer_pronostico(valor_potenciometro): # pyFirmata devuelve un valor 0.0–1.0; lo mapeamos a goles (0 a 5) goles = round(valor_potenciometro * 5) return goles def mostrar_pronostico(goles_local, goles_visitante): # Recibe el pronóstico de ambos equipos y arma el mensaje a mostrar if goles_local > goles_visitante: ganador = "ganador: Local" elif goles_visitante > goles_local: ganador = "ganador: Visitante" else: ganador = "empate" return f"Tu pronóstico: Local {goles_local} - {goles_visitante} Visitante | {ganador}" it = pyfirmata.util.Iterator(board) it.start() pot_local.enable_reporting() pot_visitante.enable_reporting() while True: valor_l = pot_local.read() or 0 valor_v = pot_visitante.read() or 0 goles_l = leer_pronostico(valor_l) goles_v = leer_pronostico(valor_v) print(mostrar_pronostico(goles_l, goles_v)) time.sleep(1)
🔄
Fijate que mostrarPronostico() / mostrar_pronostico() usa el resultado de leerPronostico() / leer_pronostico() como entrada. Una función puede alimentarse de lo que devuelve otra — así arman sistemas completos, como en el proyecto de la maqueta.

El Indicador de Puntaje — función con parámetro por defecto

El servo que movían entre 0° y 180° en Tinkercad ahora es la "aguja" que marca cuántos puntos ganaste con tu pronóstico.

Contexto TinkercadRecordá tu circuito

El servo se controla con un ángulo entre 0° y 180°, usando la librería Servo.h. Se escribe con miServo.write(angulo). Hoy escribimos la función que decide cuánto mover la aguja según los puntos que sumaste.

🤖
Función con parámetro por defecto
int marcarPuntaje(String nivel = "sin_puntos")
#include <Servo.h> Servo servoPuntaje; const int PIN_SERVO = 9; int marcarPuntaje(String nivel = "sin_puntos") { // Calcula el ángulo de la aguja según los puntos ganados con el pronóstico. // Si no se indica nada, asume que no hubo puntos. int angulo; if (nivel == "resultado_exacto") { angulo = 180; // acertaste el marcador exacto — 3 puntos } else if (nivel == "ganador") { angulo = 90; // acertaste quién gana, no el marcador — 1 punto } else { angulo = 0; // no sumaste puntos } servoPuntaje.write(angulo); Serial.println("Aguja en " + String(angulo) + " grados — puntaje: " + nivel); return angulo; } void setup() { Serial.begin(9600); servoPuntaje.attach(PIN_SERVO); } void loop() { // Tres formas de llamar a la misma función marcarPuntaje("resultado_exacto"); // acertaste todo delay(2000); marcarPuntaje("ganador"); delay(2000); marcarPuntaje(); // usa "sin_puntos" por defecto delay(2000); }
import pyfirmata import time board = pyfirmata.Arduino('COM3') servo_puntaje = board.get_pin('d:9:s') def marcar_puntaje(nivel="sin_puntos"): # Calcula el ángulo de la aguja según los puntos ganados con el pronóstico. # Si no se indica nada, asume que no hubo puntos. if nivel == "resultado_exacto": angulo = 180 # acertaste el marcador exacto — 3 puntos elif nivel == "ganador": angulo = 90 # acertaste quién gana, no el marcador — 1 punto else: angulo = 0 # no sumaste puntos servo_puntaje.write(angulo) print(f"Aguja en {angulo} grados — puntaje: {nivel}") return angulo # Tres formas de llamar a la misma función — Python también soporta valores por defecto marcar_puntaje("resultado_exacto") time.sleep(2) marcar_puntaje("ganador") time.sleep(2) marcar_puntaje() # usa "sin_puntos" por defecto time.sleep(2)
Novedad de hoy: tanto String nivel = "sin_puntos" en C++ como nivel="sin_puntos" en Python son parámetros por defecto. Si no le pasás nada al llamar la función, usa ese valor automáticamente — la idea es la misma en los dos lenguajes.

Elegir el Partido — función que recorre un fixture

El encoder que contaban "clics" en Tinkercad ahora navega la lista de partidos del Mundial para elegir cuál pronosticar.

Contexto TinkercadRecordá tu circuito

El encoder gira y genera pulsos: cada "clic" suma o resta una unidad según el sentido de giro. Acá lo usamos para mover un índice dentro del fixture (la lista de partidos) y elegir cuál pronosticar.

🔄
Función que recorre un arreglo
int elegirPartido(int pulsosEncoder[], int cantidad, int totalPartidos)
String fixture[] = { "Argentina vs Brasil", "Francia vs Alemania", "España vs Italia", "Uruguay vs Croacia" }; const int TOTAL_PARTIDOS = 4; int elegirPartido(int pulsosEncoder[], int cantidad, int totalPartidos) { // Recibe los pulsos del encoder (+1 = avanza, -1 = retrocede) // y devuelve el índice final dentro del fixture, sin salirse del rango int indice = 0; for (int i = 0; i < cantidad; i++) { indice += pulsosEncoder[i]; if (indice < 0) indice = 0; if (indice >= totalPartidos) indice = totalPartidos - 1; } return indice; } void setup() { Serial.begin(9600); // Simulación: pulsos del encoder mientras el alumno navega el fixture int pulsos[] = {1, 1, 1, -1, 1}; int cantidadPulsos = 5; int indiceElegido = elegirPartido(pulsos, cantidadPulsos, TOTAL_PARTIDOS); Serial.println("Partido elegido: " + fixture[indiceElegido]); } void loop() { // vacío — esta misión corre una sola vez en setup() }
fixture = [ "Argentina vs Brasil", "Francia vs Alemania", "España vs Italia", "Uruguay vs Croacia" ] TOTAL_PARTIDOS = len(fixture) def elegir_partido(pulsos_encoder, total_partidos): # Recibe una lista de pulsos del encoder (+1 = avanza, -1 = retrocede) # y devuelve el índice final dentro del fixture, sin salirse del rango indice = 0 for pulso in pulsos_encoder: indice += pulso if indice < 0: indice = 0 if indice >= total_partidos: indice = total_partidos - 1 return indice # Simulación: pulsos del encoder mientras el alumno navega el fixture pulsos = [1, 1, 1, -1, 1] indice_elegido = elegir_partido(pulsos, TOTAL_PARTIDOS) print(f"Partido elegido: {fixture[indice_elegido]}")
⚽
Es el mismo patrón "acumulador" del for de T1, solo que ahora el resultado se usa como índice de un arreglo/lista de Strings. En C++, cuando un arreglo es parámetro, hay que pasar también su tamaño aparte (acá cantidad); en Python no hace falta, porque las listas ya "saben" su tamaño.

Integrador — el Prode completo

Las cuatro funciones trabajando juntas en un único programa principal, como un sistema real.

🎯
Programa principal
setup() y loop() del Prode completo
// ─── LIBRERÍAS Y FUNCIONES (van arriba del setup) ────────── // elegirPartido(), leerPronostico(), mostrarPronostico(), // confirmarPronostico(), marcarPuntaje() → ya las escribiste arriba #include <Servo.h> Servo servoPuntaje; String fixture[] = {"Argentina vs Brasil", "Francia vs Alemania", "España vs Italia"}; const int TOTAL_PARTIDOS = 3; const int PIN_PULSADOR = 2; const int PIN_POT_LOCAL = A0; const int PIN_POT_VISITANTE = A1; const int PIN_SERVO = 9; void setup() { Serial.begin(9600); pinMode(PIN_PULSADOR, INPUT); servoPuntaje.attach(PIN_SERVO); } void loop() { // 1) Encoder → elegimos el partido (simulado acá con pulsos fijos) int pulsosEncoder[] = {1, 1}; int indice = elegirPartido(pulsosEncoder, 2, TOTAL_PARTIDOS); Serial.println("Partido: " + fixture[indice]); // 2) Potenciómetros → armamos el pronóstico int golesLocal = leerPronostico(analogRead(PIN_POT_LOCAL)); int golesVisitante = leerPronostico(analogRead(PIN_POT_VISITANTE)); Serial.println(mostrarPronostico(golesLocal, golesVisitante)); // 3) Pulsador → confirmamos el pronóstico String estado = confirmarPronostico(digitalRead(PIN_PULSADOR)); Serial.println(estado); // 4) Servo → una vez jugado el partido, marcamos el puntaje obtenido int golesRealLocal = 2; // resultado real (vendría de otra fuente) int golesRealVisitante = 1; if (golesLocal == golesRealLocal && golesVisitante == golesRealVisitante) { marcarPuntaje("resultado_exacto"); } else if ((golesLocal > golesVisitante && golesRealLocal > golesRealVisitante) || (golesVisitante > golesLocal && golesRealVisitante > golesRealLocal)) { marcarPuntaje("ganador"); } else { marcarPuntaje(); } delay(3000); }
import pyfirmata import time board = pyfirmata.Arduino('COM3') pin_pulsador = board.get_pin('d:2:i') pot_local = board.get_pin('a:0:i') pot_visitante = board.get_pin('a:1:i') servo_puntaje = board.get_pin('d:9:s') it = pyfirmata.util.Iterator(board) it.start() pin_pulsador.enable_reporting() pot_local.enable_reporting() pot_visitante.enable_reporting() fixture = ["Argentina vs Brasil", "Francia vs Alemania", "España vs Italia"] TOTAL_PARTIDOS = len(fixture) # (acá van las funciones: elegir_partido, leer_pronostico, mostrar_pronostico, # confirmar_pronostico, marcar_puntaje → ya las escribiste arriba) while True: # 1) Encoder → elegimos el partido (simulado acá con pulsos fijos) pulsos_encoder = [1, 1] indice = elegir_partido(pulsos_encoder, TOTAL_PARTIDOS) print(f"Partido: {fixture[indice]}") # 2) Potenciómetros → armamos el pronóstico goles_local = leer_pronostico(pot_local.read() or 0) goles_visitante = leer_pronostico(pot_visitante.read() or 0) print(mostrar_pronostico(goles_local, goles_visitante)) # 3) Pulsador → confirmamos el pronóstico lectura = pin_pulsador.read() estado = confirmar_pronostico(1 if lectura else 0) print(estado) # 4) Servo → una vez jugado el partido, marcamos el puntaje obtenido goles_real_local = 2 # resultado real (vendría de otra fuente) goles_real_visitante = 1 if goles_local == goles_real_local and goles_visitante == goles_real_visitante: marcar_puntaje("resultado_exacto") elif (goles_local > goles_visitante and goles_real_local > goles_real_visitante) or \ (goles_visitante > goles_local and goles_real_visitante > goles_real_local): marcar_puntaje("ganador") else: marcar_puntaje() time.sleep(3)
💡 Esto es el mismo esqueleto del proyecto "Red Multi-Punto" de la maqueta: funciones chicas y específicas + un loop() (o while True) corto que las va llamando en orden. En Arduino corre directo en el microcontrolador; en Python corre en la PC y se comunica con la placa por pyFirmata — el puente que ya usaron en T2.
